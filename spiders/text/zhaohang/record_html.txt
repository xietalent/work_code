        function getBeginDate(type, endDate, Span) {
            if (type ==  "day")
            {
                return getBeginDateByDay(endDate, Span);
            }
            else if (type == "month")
            {
                return getBeginDateByMonth(endDate, Span)
            }
        }

        function getBeginDateByDay(endDate, daySpan) {
            var date = new Date(endDate.replace(/(\d{4})(\d{2})(\d{2})/, '$1/$2/$3'));
            var day = date.getDate() + 1 - daySpan;
            var month = date.getMonth() + 1;
            var year = date.getFullYear();
                        
            month = (day <= 0) ? month - 1 : month;
            year = (month <= 0) ? year - 1 : year;
            month = (month <= 0) ? month + 12 : month;

            if (day <= 0)
            {
                var lastMonOfdayCout = 31;
                if (month == 2) 
                {
                    lastMonOfdayCout =  (((year % 4 === 0) && (year % 100 !== 0)) || (year % 400 === 0)) ? 29 : 28;
                }
                if (month == 4 || month == 6 || month == 9 || month == 11)
                {
                    lastMonOfdayCout = 30;
                }
                day = day + lastMonOfdayCout;
            }

            day = (day < 10 ? '0' : '') + day;
            month = (month < 10 ? '0' : '') + month;
            return year + month + day;
        }

        function getBeginDateByMonth(endDate, monthSpan) {
            var date = new Date(endDate.replace(/(\d{4})(\d{2})(\d{2})/, '$1/$2/$3'));
            var day = date.getDate();
            var month = date.getMonth() + 1 - monthSpan;
            var year = date.getFullYear();

            year = (month <= 0) ? year - 1 : year;
            month = (month <= 0) ? month + 12 : month;
            day = (day > 28 && month == 2) ? ((((year % 4 === 0) && (year % 100 !== 0)) || (year % 400 === 0)) ? 29 : 28) : day;
            day = (day == 31 && (month == 4 || month == 6 || month == 9 || month == 11)) ? 30 : day;

            day = (day < 10 ? '0' : '') + day;
            month = (month < 10 ? '0' : '') + month;
            return year + month + day;
        }
